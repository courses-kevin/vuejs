### Binding Descendant (Argument)

Pour descendre une donnée dans l'arbre des composants

```html
<div v-bind:id="dynamicId"></div>
```

Sucre:

```html
<div :id="dynamicId"></div>
```

---

### Binding Montant (Events)

Pour remonter une donnée dans l'arbre des composants

```html
<a v-on:click="doSomething"> ... </a>
```

Sucre:

```html
<a @click="doSomething"> ... </a>
```

---

### Modifier (Exemple event)

`event.preventDefault()`

```html
<form @submit.prevent="onSubmit">...</form>
```

#### Types complexes (array, object, collection, ...)

```js
import { reactive } from "vue";

export default {
  // setup défini par VueJS donc non renommable
  setup() {
    const state = reactive({ count: 0 });

    function increment() {
      state.count++;
    }

    return {
      state,
      increment,
    };
  },
};
```

\+

```html
<button @click="increment">{{ state.count }}</button>
```

---

#### Sucre du `setup`

```html
// Indication du setup dans la balise script
<script setup>
  import { reactive } from "vue";

  const state = reactive({ count: 0 });

  function increment() {
    state.count++;
  }
</script>

<template>
  <button @click="increment">{{ state.count }}</button>
</template>
```

---

#### Types primitifs (string, boolean, number, ...)

```html
<script setup>
  import { ref } from "vue";

  const count = ref(0);

  function increment() {
    count.value++;
  }
</script>

<template>
  <button @click="increment">
    {{ count }}
    <!-- no .value needed -->
  </button>
</template>
```
